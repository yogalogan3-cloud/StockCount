        // PRE-CONFIGURED CONNECTION KEYS

        // Initialize Supabase Connection

        const supabase = supabase.createClient(SUPABASE_URL, SUPABASE_KEY);

 

        // Handle Password Screen Check

        function checkPassword() {

            const enteredValue = document.getElementById('staff-password').value;

            if (enteredValue === APP_PASSWORD) {

                document.getElementById('password-gate').style.display = 'none';

                document.getElementById('app-interface').style.display = 'block';

            } else {

                document.getElementById('login-error').innerText = "Incorrect Password. Try again.";

            }

        }

 

        // Handle Form Submission to your cloud database

        document.getElementById('stock-form').addEventListener('submit', async (e) => {

            e.preventDefault();

            const submitBtn = document.getElementById('submit-btn');

            const statusMsg = document.getElementById('submit-status');

           

            submitBtn.disabled = true;

            submitBtn.innerText = "Saving Data...";

            statusMsg.style.color = "orange";

            statusMsg.innerText = "Connecting to database...";

 

            // Gather values directly from the app screen

            const submissionData = {

                staff_name: document.getElementById('staff_name').value,

                milk_cartons: parseInt(document.getElementById('milk_cartons').value) || 0,

                coffee_bags: parseInt(document.getElementById('coffee_bags').value) || 0

            };

 

            // Send data directly to your table

            const { data, error } = await supabase

                .from('stock_counts')

                .insert([submissionData]);

 

            if (error) {

                console.error("Database Error:", error);

                statusMsg.style.color = "red";

                statusMsg.innerText = "Error saving data: " + error.message;

                submitBtn.disabled = false;

                submitBtn.innerText = "Submit to Cloud Dashboard";

            } else {

                statusMsg.style.color = "green";

                statusMsg.innerText = "Success! Counts saved securely.";

                document.getElementById('stock-form').reset();

                submitBtn.disabled = false;

                submitBtn.innerText = "Submit to Cloud Dashboard";

            }

        });
